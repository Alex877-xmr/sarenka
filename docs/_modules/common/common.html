[docs]    @staticmethod
    def parse_to_list(data):
        """
        Parsuje plik do listy
        """
        if isinstance(data, list):
            output = data

        if isinstance(data, str):
            with open(data, "r") as f:
                output = f.readlines()
            output = [i.rstrip() for i in output]

        return output
